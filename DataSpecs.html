        defmodule Person do
  use DataSpecs

  @enforce_keys [:name, :surname]
  defstruct @enforce_keys ++ [:birth_date]

  @type t :: %__MODULE__{
              name: String.t(),
              surname: String.t(),
              birth_date: nil | Date.t(),
            }
end
